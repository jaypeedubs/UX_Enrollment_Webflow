── REQUESTS ──
  loadPrograms
    Type   : Supabase — Get rows — table: programs
    Filter : status = "active"
    Select : id, name, deadline, program_questions
    Run    : on page load
  loadDraft
    Type   : Supabase — Get rows — table: applications
    Filter : applicant_id = auth.uid(), status = "draft"
    Select : id, program_id, status, cv_url, locked_fields, programs(name)
    Limit  : 1
    Run    : on page load
    On success with data: pre-fill form fields, set applicationId, set locks
  saveDraft
    Type   : Supabase — Upsert row — table: applications
    Body: {
      id          : @applicationId (if exists) or omit for auto-generate,
      applicant_id: @Wized.user.id,
      program_id  : @programId,
      status      : "draft",
      locked_fields: { program: true, first_name: true, last_name: true }
    }
    Update user metadata: first_name, last_name via Wized.auth.updateUser()
    On success: set applicationId from response; insert draft_saved event:
      POST /rest/v1/application_events
        body: { application_id: @applicationId, event_type: "draft_saved" }
  uploadCV
    Type   : Supabase Storage — Upload file
    Bucket : cvs
    Path   : @Wized.user.id + "/" + filename
    File   : @[wized=cv-file-input].files[0]
    On success: PATCH applications SET cv_url = storage_path WHERE id = @applicationId
  submitApplication
    Type   : Supabase — Update rows — table: applications
    Filter : id = @applicationId, status = "draft"
    Data   : { status: "submitted", submitted_at: new Date().toISOString(),
               locked_fields: { all: true } }
    Then   : POST /rest/v1/application_events
             body: { application_id: @applicationId, event_type: "submitted" }
    On success → redirect to /dashboard
